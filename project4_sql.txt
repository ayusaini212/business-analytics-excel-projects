SELECT COUNT(DISTINCT "Order ID") AS total_orders,
       SUM(Profit) AS  total_profit,
	   SUM(sales) AS total_revenue,
	   ROUND(SUM(profit) * 100 / SUM(sales), 1) AS margin_percent,
	   ROUND(AVG(discount) * 100, 1) AS avg_discount_percentage
	   FROM sales;



SELECT Region,
      COUNT(DISTINCT "order id") AS total_orders,
	  SUM(sales) AS total_revenue,
      SUM(profit) AS total_profit,
	  SUM(profit) * 100 / SUM(sales) AS margin_percent
	  FROM sales
GROUP BY Region
ORDER BY total_profit DESC;	  


SELECT Category , Sub_Category,
      COUNT(DISTINCT "order id") AS total_orders,
	  SUM(sales) AS total_revenue,
      SUM(profit) AS total_profit,
	  SUM(profit) * 100 / SUM(sales) AS margin_percent
	  FROM sales
GROUP BY Category, Sub_Category
ORDER BY margin_percent DESC;




 SELECT Ship_Mode,
       COUNT(*) AS total_orders,
	    ROUND(AVG(julianday(Ship_Date) - julianday(Order_Date)), 1) AS avg_delivery_days,
	   ROUND(SUM(sales), 0) AS revenue 
	   FROM sales
GROUP BY Ship_Mode
ORDER BY avg_delivery_days;




SELECT 
      CASE 
	  WHEN Discount = 0 THEN 'no discount'
	  WHEN Discount <= 0.2 THEN 'low (up to 20%)'
	  WHEN Discount <= 0.4 THEN 'medium (20-40%)'
	  ELSE 'high (above 40%)'
	  END AS discount_tier,
	 COUNT(*) AS total_orders,
     SUM(profit) AS total_profit,
     SUM(profit * 100) / SUM(sales) AS profit_margin,
	 AVG(profit) AS avg_profit
	 FROM sales
	 GROUP BY discount_tier
	 ORDER BY discount_tier DESC;



SELECT Product_Name,
       category,
       SUM(profit) AS total_profit,
	   SUM(sales) AS total_revenue,
	   SUM(quantity) AS units_sold
	   FROM sales
	   GROUP BY Product_Name, Category
	   ORDER BY total_profit DESC
	   LIMIT 10;
	   
	   


SELECT Returned,
       COUNT(*) AS total_orders,
	   SUM(Profit) AS total_profit,
	   SUM(sales) AS total_revenue
	   FROM sales
	   GROUP BY Returned;
